View Module
interface DocumentModule {
    handler: "@pi-r/android";
    extensions?: string[];
    versions?: StringMap;
    settings?: {
        broadcast_id?: string | string[];
        users?: Record<string, Record<string, unknown>>;
        extensions?: {
            task?: {
                exec?: {
                    uid?: number | string;
                    gid?: number | string;
                };
                command?: string;
            };
        };
        language?: {
            gradle?: "java" | "kotlin" | "java+kotlin";
        };
        directory?: {
            template?: string;
        };
    };
    permission: PermittedDirectories;
}
